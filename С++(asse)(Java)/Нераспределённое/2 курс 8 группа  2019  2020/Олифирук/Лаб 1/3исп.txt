package homeworkex3;
import java.util.Scanner;
/**
 *
 * @author Maxim
 */
public class program 
{ 
    public static double sin(double x)
    {
        double sin=0;
        double znam=1;
        double drob=x/znam;
        double e=0.001;
        while(Math.abs(1/znam)>e)
        {
           sin=sin+drob;
           drob=drob*x*x*(-1)/((znam+1)*(znam+2));
           znam=znam+2;
        }
        return sin;
    }
    
    public static void main(String[] args) 
    {
        Scanner sc=new Scanner(System.in);
        System.out.println("Введите аргумент синуса: ");
        double x=sc.nextDouble();
        double sin=sin(x);
        System.out.println("Ответ полученный с помощью формулы Тейлора: "+sin);
        double sin1=Math.sin(x);
        System.out.println("Ответ полученный с помощью функции Math.sin(x): "+sin1);
    }
    
}
