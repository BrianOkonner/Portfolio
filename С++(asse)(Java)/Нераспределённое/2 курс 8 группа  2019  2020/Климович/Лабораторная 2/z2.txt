import java.util.Arrays;
import java.util.Comparator;

class revcomp2 implements Comparator<Integer> {
    @Override
    public int compare(Integer t1, Integer t2) {
        if (t1 > t2) return -1;
        if (t1 < t2) return 1;
        return 0;
    }
}

public class appArrays2 {   //в заданном массиве заменить за один просмотр все возрастающие группы чисел
    public static void main(String[] args) {    //на те же группы  в обратном порядке

        Integer[] a = {7, 8, 9, 1, 4, 5};
        int p = 0;
        for (int i = 1; i < a.length; i++) {
            if (a[i] < a[i - 1]) {
                Arrays.sort(a, p, i, new revcomp2());
                p = i;
            }
        }
        System.out.println(Arrays.toString(a));
    }
}
