    public static boolean palindrom(int n)
    {
        int k=n;
        int perev=0;
        while(n!=0)
        {
            perev=perev*10;
            perev=perev+n%10;
            n=n/10;   
        }
        if(k==perev)
        {
            return true;
        }
        else
        {
            return false;
        }
    }

           }

       }
