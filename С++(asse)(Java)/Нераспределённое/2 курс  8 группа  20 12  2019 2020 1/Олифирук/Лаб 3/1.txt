package lab3ex1;
import java.util.Scanner;
import java.util.Arrays;
import java.io.*;
/**
 *
 * @author Maxim
    В матрице найти наибольшее количество подряд идущих чисел в строке
 */
public class Lab3Ex1 
{

    public static void main(String[] args) 
    {
        try
        {
            FileInputStream InputF=new FileInputStream("Input.txt");
            Scanner sc=new Scanner(InputF);
            int a=sc.nextInt();
            int b=sc.nextInt();
            int[][]Mas=new int[a][b];
            for(int i=0;i<a;i++)
            {
                for(int j=0;j<b;j++)
                {
                    Mas[i][j]=sc.nextInt();
                }
            }
            
            int sum=1;
            int max=0;
            int maxstroki=0;
            int numstroki=0;
            for(int i=0;i<a;i++)
            {
                int j=0;
                while(j!=b-1)
                {
                    if(Mas[i][j]==Mas[i][j+1])
                    {
                        sum++; 
                        j++;
                    }
                    else
                    {
                        max=sum;
                        j++;
                        sum=1;
                    }
                    
                     if(max>maxstroki)
                    {
                        maxstroki=max;
                        numstroki=i;
                    }
                }
                
               
            }
            
            System.out.println("numstroki="+numstroki);
        }
        catch(IOException e)
        {
            System.out.println("Файл не найден");
        }
    }
    
}
