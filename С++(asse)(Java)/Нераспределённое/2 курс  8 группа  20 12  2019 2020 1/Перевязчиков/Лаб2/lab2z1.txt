class cmpsumzn implements Comparator<Integer>
{
    @ Override public int compare(Integer t11,Integer t22)
    {int t1=sumzn(t11);
    int t2=sumzn(t22);
    if(t1==t2)
        return 0;
    if(t1>t2)
        return -1;
    return 1;
    }
    public static int sumzn(int t)
    {
        int i;
        int tenvstep=1,sumn=0,copy=t;
        while(copy!=0)
        {
            tenvstep*=10;
            sumn+=copy%10;
            copy/=10;
        }
        return sumn;
    }
}
